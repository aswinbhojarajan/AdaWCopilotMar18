import React, { useState } from 'react';
import svgPaths from '../../imports/svg-gkxr0z6k1p';
import imgSignIn from "figma:asset/7cd51e151c476d332035876bab32970565a63a2c.png";

interface SignInScreenProps {
  onSignIn?: (email: string, password: string) => void;
  authError?: string | null;
}

/**
 * SignInScreen Component
 * 
 * Authentication screen for Ada users.
 * Features floating labels, password visibility toggle,
 * clear icons, and custom error messages.
 */
export function SignInScreen({ onSignIn, authError }: SignInScreenProps) {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [showPassword, setShowPassword] = useState(false);
  const [focusedField, setFocusedField] = useState<string | null>(null);
  const [error, setError] = useState<string>('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    
    // Clear previous error
    setError('');
    
    // Validate email format
    if (email && !email.includes('@')) {
      setError('Please add "@" in the email address.');
      return;
    }
    
    // Check if both fields are filled
    if (!email || !password) {
      setError('Please enter both email and password.');
      return;
    }
    
    if (onSignIn) {
      onSignIn(email, password);
    }
  };

  // Display auth error from parent or local error
  const displayError = authError || error;

  const isEmailActive = focusedField === 'email' || email.length > 0;
  const isPasswordActive = focusedField === 'password' || password.length > 0;

  return (
    <div className="relative size-full min-h-screen bg-[#f7f6f2]">
      {/* Background with decorative image */}
      <div aria-hidden="true" className="absolute inset-0 pointer-events-none">
        <div className="absolute bg-[#f7f6f2] inset-0" />
        <div className="absolute inset-0 opacity-25 overflow-hidden">
          <img alt="" className="absolute h-full left-[-107.63%] max-w-none top-0 w-[315.26%]" src={imgSignIn} />
        </div>
      </div>

      {/* Ada Logo & Wordmark */}
      <div className="absolute left-1/2 -translate-x-1/2 top-[140px] flex flex-col items-center">
        {/* Ada Logo SVG */}
        <svg width="182" height="69" viewBox="0 0 182 69" fill="none" xmlns="http://www.w3.org/2000/svg">
          <path d="M64.1447 56L84.2294 11.4371H86.4889L106.574 56H98.7908L94.1462 44.7651C93.2675 42.6311 92.7026 42.2545 89.7527 42.2545H75.5051L70.6723 56H64.1447ZM77.4508 37.7355C77.4508 38.9908 78.3295 39.4929 80.338 39.4929H91.8867L83.6645 20.0986L79.2082 31.7729C77.953 34.9739 77.4508 36.543 77.4508 37.7355ZM121.882 56.9415C113.66 56.9415 108.136 51.0416 108.136 42.0035C108.136 31.7101 115.543 23.99 125.397 23.99C129.853 23.99 133.556 25.4964 136.004 28.1325L135.313 12.0648H141.715V56H135.941L135.816 49.6608C132.866 54.1171 127.719 56.9415 121.882 56.9415ZM124.706 53.1756C131.234 53.1756 136.004 47.5268 136.004 39.6812C136.004 31.9612 131.485 26.94 124.832 26.94C118.304 26.94 114.538 31.5218 114.538 39.6812C114.538 47.9661 118.493 53.1756 124.706 53.1756ZM158.365 56.9415C152.591 56.9415 149.139 53.866 149.139 48.7821C149.139 44.3885 151.649 41.7524 158.742 38.8025L164.579 36.3547C167.78 35.0366 168.721 33.9069 168.721 31.5218C168.721 28.5719 166.713 27.1283 163.386 27.1283C157.989 27.1283 153.595 30.8942 151.273 36.9823V26.4379C154.913 24.9315 159.934 23.99 164.077 23.99C171.734 23.99 175.249 27.191 174.935 33.8441L174.307 48.4682C174.182 50.665 175.061 52.0458 176.567 52.0458C178.073 52.0458 179.015 51.6065 180.584 50.1629C180.584 54.0543 178.073 56.5649 174.119 56.5649C170.855 56.5649 169.098 54.8702 168.91 51.4809C166.901 54.933 163.073 56.9415 158.365 56.9415ZM160.813 52.9245C164.893 52.9245 168.721 49.9746 168.721 46.7736V37.0451L161.817 40.4344C157.11 42.7567 155.541 44.5768 155.541 47.7778C155.541 50.9161 157.549 52.9245 160.813 52.9245Z" fill="#441316"/>
          <path d="M44.0877 55.4589C43.0973 56.4493 41.6186 56.9989 39.636 57.0121C37.6461 57.0253 35.2342 56.493 32.5421 55.4162C27.1618 53.2643 20.8808 49.0222 15.0257 43.1672C9.1711 37.3125 4.9303 31.0321 2.77816 25.6523C1.70157 22.9602 1.16925 20.5481 1.18224 18.5582C1.19543 16.5755 1.74507 15.0969 2.73548 14.1065C3.72589 13.1165 5.20326 12.5679 7.18574 12.5549C9.17563 12.5417 11.589 13.0724 14.2811 14.1494C19.6609 16.3013 25.9401 20.5437 31.7947 26.3982C37.6496 32.2532 41.8931 38.533 44.045 43.9133C45.1218 46.6054 45.6527 49.0187 45.6395 51.0086C45.6263 52.9907 45.0777 54.4687 44.0877 55.4589Z" stroke="#441316" strokeWidth="2.36402"/>
          <path d="M3.34281 55.4589C4.33301 56.4493 5.81183 56.9989 7.79451 57.0121C9.7842 57.0253 12.1963 56.493 14.8884 55.4162C20.2685 53.2643 26.5497 49.0222 32.4047 43.1672C38.2594 37.3125 42.5002 31.0321 44.6521 25.6523C45.7289 22.9602 46.2612 20.5481 46.248 18.5582C46.235 16.5755 45.6852 15.0969 44.695 14.1065C43.7046 13.1165 42.2272 12.5679 40.2447 12.5549C38.2546 12.5417 35.8415 13.0724 33.1492 14.1494C27.7693 16.3013 21.4902 20.5437 15.6358 26.3982C9.7809 32.2532 5.53742 38.533 3.38548 43.9133C2.30869 46.6054 1.77782 49.0187 1.79101 51.0086C1.80421 52.9907 2.35281 54.4687 3.34281 55.4589Z" stroke="#441316" strokeWidth="2.36402"/>
        </svg>
      </div>

      {/* SIGN IN Label */}
      <p className="-translate-x-1/2 absolute font-['DM_Sans:SemiBold',sans-serif] leading-[18px] left-[calc(50%-1px)] not-italic text-[#431315] text-[12px] text-center top-[510px] tracking-[0.8px] uppercase whitespace-nowrap">SIGN IN</p>

      {/* Tagline */}
      <p className="-translate-x-1/2 absolute font-['Crimson_Pro:Light',sans-serif] font-light leading-[normal] left-[calc(50%-1px)] text-[#555] text-[22px] text-center top-[533px] tracking-[-0.24px] whitespace-nowrap">Your personalized wealth intelligence</p>

      {/* Sign In Form */}
      <form onSubmit={handleSubmit} noValidate>
        {/* Email Input */}
        <div className={`absolute bg-white content-stretch flex h-[58px] items-center justify-between left-[20px] px-[24px] rounded-[50px] top-[587px] w-[390px]`}>
          <div className={`content-stretch flex flex-col relative shrink-0 flex-1 ${isEmailActive ? 'items-start justify-start' : 'items-start justify-center'}`}>
            {/* Floating label */}
            <div className={`content-stretch flex items-center overflow-clip transition-all duration-200 ${isEmailActive ? 'opacity-50' : 'opacity-50'}`}>
              <p className={`font-['DM_Sans:Regular',sans-serif] leading-[normal] not-italic text-black transition-all duration-200 pointer-events-none ${isEmailActive ? 'h-[15px] text-[12px]' : 'h-[21px] text-[16px]'}`}>
                Email
              </p>
            </div>
            
            {/* Input value */}
            {isEmailActive && (
              <div className="content-stretch flex h-[21px] items-center overflow-clip">
                <input
                  type="email"
                  value={email}
                  onChange={(e) => setEmail(e.target.value)}
                  onFocus={() => setFocusedField('email')}
                  onBlur={() => setFocusedField(null)}
                  className="font-['DM_Sans:Regular',sans-serif] leading-[normal] not-italic text-[#555] text-[16px] w-full bg-transparent border-none outline-none p-0"
                />
              </div>
            )}
            
            {/* Clickable overlay when not active */}
            {!isEmailActive && (
              <div 
                className="absolute inset-0 cursor-text"
                onClick={() => {
                  setFocusedField('email');
                  setTimeout(() => {
                    const input = document.querySelector('input[type="email"]') as HTMLInputElement;
                    if (input) input.focus();
                  }, 0);
                }}
              />
            )}
          </div>

          {/* Clear icon */}
          {email && (
            <button
              type="button"
              onClick={() => setEmail('')}
              className="flex items-center justify-center shrink-0 size-[20px] cursor-pointer hover:opacity-80 transition-opacity"
              aria-label="Clear email"
            >
              <svg className="block size-full" fill="none" preserveAspectRatio="none" viewBox="0 0 20 20">
                <g>
                  <rect fill="black" fillOpacity="0.1" height="20" rx="10" width="20" />
                  <path d={svgPaths.p701bf80} fill="#555555" />
                </g>
              </svg>
            </button>
          )}
        </div>

        {/* Password Input */}
        <div className={`absolute bg-white content-stretch flex h-[58px] items-center justify-between left-[20px] px-[24px] rounded-[50px] top-[657px] w-[390px]`}>
          <div className={`content-stretch flex flex-col relative shrink-0 flex-1 ${isPasswordActive ? 'items-start justify-start' : 'items-start justify-center'}`}>
            {/* Floating label */}
            <div className={`content-stretch flex items-center overflow-clip transition-all duration-200 ${isPasswordActive ? 'opacity-50' : 'opacity-50'}`}>
              <p className={`font-['DM_Sans:Regular',sans-serif] leading-[normal] not-italic text-black transition-all duration-200 pointer-events-none ${isPasswordActive ? 'h-[15px] text-[12px]' : 'h-[21px] text-[16px]'}`}>
                Password
              </p>
            </div>
            
            {/* Input value */}
            {isPasswordActive && (
              <div className="content-stretch flex h-[21px] items-center overflow-clip">
                <input
                  type={showPassword ? "text" : "password"}
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  onFocus={() => setFocusedField('password')}
                  onBlur={() => setFocusedField(null)}
                  className="font-['DM_Sans:Regular',sans-serif] leading-[normal] not-italic text-[#555] text-[16px] w-full bg-transparent border-none outline-none p-0"
                />
              </div>
            )}
            
            {/* Clickable overlay when not active */}
            {!isPasswordActive && (
              <div 
                className="absolute inset-0 cursor-text"
                onClick={() => {
                  setFocusedField('password');
                  setTimeout(() => {
                    const input = document.querySelector('input[type="password"]') as HTMLInputElement;
                    if (input) input.focus();
                  }, 0);
                }}
              />
            )}
          </div>

          {/* Visibility toggle icon */}
          {password && (
            <button
              type="button"
              onClick={() => setShowPassword(!showPassword)}
              className="flex items-center justify-center shrink-0 size-[24px] cursor-pointer hover:opacity-80 transition-opacity"
              aria-label={showPassword ? "Hide password" : "Show password"}
            >
              <svg className="block size-full" fill="none" preserveAspectRatio="none" viewBox="0 0 24 24">
                <g>
                  <path 
                    d={showPassword ? svgPaths.p50f5f80 : svgPaths.p19a84880} 
                    fill="#555555" 
                  />
                </g>
              </svg>
            </button>
          )}
        </div>

        {/* Log in Button */}
        <button 
          type="submit"
          className="absolute bg-[#431315] content-stretch flex flex-col h-[58px] items-center justify-center left-[20px] px-[24px] rounded-[50px] top-[755px] w-[390px] cursor-pointer hover:brightness-110 active:scale-[0.98] transition-all duration-200"
        >
          <p className="font-['DM_Sans:Regular',sans-serif] leading-[normal] not-italic text-[16px] text-center text-white whitespace-nowrap">Log in</p>
        </button>
      </form>

      {/* Error Message */}
      {displayError && (
        <div className="-translate-x-1/2 absolute bg-[#fef3c7] left-[calc(50%+0.5px)] rounded-[8px] top-[865px] w-[347px] animate-in fade-in-0 duration-200">
          <div className="content-stretch flex flex-col items-start overflow-clip px-[12px] py-[8px] relative rounded-[inherit] w-full">
            <div className="content-stretch flex items-center justify-between relative shrink-0 w-full">
              <div className="content-stretch flex gap-[8px] items-center relative shrink-0">
                <div className="overflow-clip relative shrink-0 size-[24px]">
                  <div className="absolute left-[1.89px] size-[20px] top-[2.22px]">
                    <svg className="absolute block size-full" fill="none" preserveAspectRatio="none" viewBox="0 0 20 20">
                      <g clipPath="url(#clip0_signin_error)" id="exclamation-circle-solid 1">
                        <path d={svgPaths.p13d8780} fill="#78350F" id="Vector" />
                      </g>
                      <defs>
                        <clipPath id="clip0_signin_error">
                          <rect fill="white" height="20" width="20" />
                        </clipPath>
                      </defs>
                    </svg>
                  </div>
                </div>
                <p className="font-['DM_Sans:Medium',sans-serif] leading-[normal] not-italic relative shrink-0 text-[#78350f] text-[16px] whitespace-nowrap">
                  {displayError}
                </p>
              </div>
            </div>
          </div>
          <div aria-hidden="true" className="absolute border border-[rgba(0,0,0,0.1)] border-solid inset-0 pointer-events-none rounded-[8px]" />
        </div>
      )}
    </div>
  );
}
